a)	When Switch 0 is pressed,  relay and buzzer is turned ON.
b)	When Switch 1 is pressed, relay and buzzer is turned OFF & Flash LEDâ€™s. 

#include <p18f4550.h> 
#include "vector_relocate.h"
void delay(unsigned int time)
{ 
unsigned int i,j; 
for(i=0;i<time;i++)
{
for(j=0;j<710;j++);
}
}
void main(void)
{
unsigned char val=0;
INTCON2bits.RBPU = 0; 
ADCON1=0x0F;
TRISD = 0x00; 
TRISBbits.TRISB0=1; 
TRISBbits.TRISB1=1; 
TRISCbits.TRISC1=0; 
TRISCbits.TRISC2=0; 
PORTCbits.RC1=0; 
PORTCbits.RC2=0;
while(1)
{
if(PORTBbits.RB1==0)
{
val=1;
}
if(PORTBbits.RB0==0)
{
val=2;
}
if(val==1)
{
PORTCbits.RC1=1; 
PORTCbits.RC2=1; 
}
if(val==2)
{
PORTCbits.RC1=0; 
PORTCbits.RC2=0; 
PORTD = 0x55;
delay(100); 
PORTD = 0xAA;
delay(100); 
}
}
}
