a)	Write a program to interface stepper motor to 8051 and rotate it clock-wise / anti clockwise direction continuously.
A] Clockwise direction
#include<reg51.h>
void msdelay(unsigned int);
void main(void)
{
  while(1)
  {
          P1 = 0X06;
          msdelay(500);
          P1 = 0X0C;
          msdelay(500);
          P1 = 0X09;
          msdelay(500);
          P1 = 0X03;
          msdelay(500);
  }
}
void msdelay(unsigned int time)
{ 
  unsigned int i,j;
  for(i=0;i<time;i++)
  {
    for(j=0;j<1275;j++)
    {}
  }
}
