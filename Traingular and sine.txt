Write a program to interface DAC 808 to 8051. Apply the digital input to obtain a triangular wave & sine wave.
a)	Triangular wave
#include<reg51.h>
void main(void)
{
 unsigned int x,y;
 while(1)
 {
   for(x=0;x<255;x++)
   {
     P1 = x;
   }
   for(y=255;y>0;y--)
   {
     P1 = y;
   }
 }
}

b)	Sine wave
#include<reg51.h>
void main(void)
{
 unsigned char sinvalue[12] = {128,192,238,255,238,192,128,64,17,0,17,64};
 unsigned char i;
 while(1)
 { 
   for(i=0;i<12;i++)
   {
     P1 = sinvalue[i];
   }
 }
}
